interface EnhancedRoutingScore {
  skillMatch: number;          // 0-1: Skill alignment
  availability: number;        // 0-1: Agent availability
  reputation: number;          // 0-1: Trust score
  computeFit: number;          // 0-1: Compute budget fit
  costEfficiency: number;      // 0-1: Cost per compute unit
  
  // Weighted calculation
  totalScore: number;          // Σ(metric × weight)
}

function calculateComputeFit(
  task: Task,
  agent: Agent
): number {
  const required = task.estimatedCompute;
  const available = agent.computeBudget.available;
  
  if (available >= required * 1.5) return 1.0;  // Plenty of budget
  if (available >= required) return 0.8;        // Sufficient
  if (available >= required * 0.7) return 0.5;  // Tight fit
  if (agent.computeBudget.borrowing.enabled) {
    return 0.3;  // Can borrow
  }
  return 0.0;  // Insufficient
}

function calculateRoutingScore(
  task: Task,
  agent: Agent,
  weights: Weights
): number {
  return (
    weights.skillMatch * calculateSkillMatch(task, agent) +
    weights.availability * agent.availability +
    weights.reputation * agent.reputation +
    weights.computeFit * calculateComputeFit(task, agent) +
    weights.costEfficiency * calculateCostEfficiency(task, agent)
  );
}

function calculateMarketPrice(
  market: MarketDynamics
): number {
  const utilizationRate = market.supply.utilizationRate;
  const demandPressure = market.demand.activeRequests / market.supply.totalCompute;
  
  let price = market.pricing.basePrice;
  
  // High utilization increases price
  if (utilizationRate > 0.8) {
    price *= 1.5;
  } else if (utilizationRate > 0.6) {
    price *= 1.2;
  }
  
  // High demand increases price
  if (demandPressure > 1.0) {
    price *= (1 + demandPressure * 0.3);
  }
  
  return price;
}
